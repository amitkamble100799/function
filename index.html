Q7. Write a function that accepts parameter n and returns factorial of n

Ans:

        function factorial(n){
            let number = 1;
            if(n == 0 || n == 1){
                return number;
            }
            else if(n>1){
                for(var i=n; i>=1; i--){
                    number = number*i;
                }
                return number;
            }
           
        }
        let n = 6;
        number = factorial(n)
        console.log('Factorial of ' + n + ' : ' + number);
